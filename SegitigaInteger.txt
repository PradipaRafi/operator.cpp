#include <iostream>
#include <algorithm>
using namespace std;

void tentukanSegitiga(int a, int b, int c) {
    if (a <= 0 || b <= 0 || c <= 0) {
        cout << "Tidak ada segitiga dapat dibangun." << endl;
        return;
    }

    int sisi[] = {a, b, c};
    sort(sisi, sisi + 3);

    int x = sisi[0], y = sisi[1], z = sisi[2];

    if (z >= x + y) {
        cout << "Tidak ada segitiga dapat dibangun." << endl;
    } else if (x == y && y == z) {
        cout << "Segitiga Sama Sisi." << endl;
    } else if (x == y || y == z || x == z) {
        cout << "Segitiga Sama Kaki." << endl;
    } else if (z * z == x * x + y * y) {
        cout << "Segitiga Siku-Siku." << endl;
    } else {
        cout << "Segitiga Bebas." << endl;
    }
}

int main() {
    int a, b, c;
    cout << "Masukkan tiga bilangan bulat: ";
    cin >> a >> b >> c;

    tentukanSegitiga(a, b, c);

    return 0;
}
