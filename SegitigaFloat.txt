#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

bool hampirSama(double a, double b) {
    return abs(a - b) / max(a, b) <= 0.01; // Ketelitian 1%
}

void tentukanSegitiga(double a, double b, double c) {
    if (a <= 0 || b <= 0 || c <= 0) {
        cout << "Tidak ada segitiga dapat dibangun." << endl;
        return;
    }

    double sisi[] = {a, b, c};
    sort(sisi, sisi + 3);

    double x = sisi[0], y = sisi[1], z = sisi[2];

    if (z >= x + y) {
        cout << "Tidak ada segitiga dapat dibangun." << endl;
    } else if (hampirSama(x, y) && hampirSama(y, z)) {
        cout << "Segitiga Sama Sisi." << endl;
    } else if (hampirSama(x, y) || hampirSama(y, z) || hampirSama(x, z)) {
        cout << "Segitiga Sama Kaki." << endl;
    } else if (hampirSama(z * z, x * x + y * y)) {
        cout << "Segitiga Siku-Siku." << endl;
    } else {
        cout << "Segitiga Bebas." << endl;
    }
}

int main() {
    double a, b, c;
    cout << "Masukkan tiga bilangan pecahan: ";
    cin >> a >> b >> c;

    tentukanSegitiga(a, b, c);

    return 0;
}
